**Prototype** is a creational design pattern that lets you copy existing objects without making your code dependent on their classes. 
problem ? Say you have an object, and you want to create an exact copy of it. How would you do it? First, you have to create a new object of the same class. Then you have to go through all the fields of the original object and copy their values over to the new object. 

just have a .clone() in the class itself. For faster cloning wali shit. 

```
Instead of:  new Object() → set 50 fields → done
Prototype:   clone(existing) → tweak 2 fields → done
```

---
 WHY tho? 

- Object creation is **expensive** (DB calls, heavy computation, complex setup)
- You need **many similar objects** with slight variations
- You want to avoid subclass explosion just to configure objects differently

## Python's Built-in Tools

Python gives you two cloning levels out of the box via the `copy` module:

```python
import copy

# Shallow copy — nested objects are SHARED
clone = copy.copy(original)

# Deep copy — nested objects are FULLY INDEPENDENT
clone = copy.deepcopy(original)
```

## 1. Basic Prototype

```python
import copy

class Character:
    def __init__(self, name, health, armor, weapons):
        self.name = name
        self.health = health
        self.armor = armor
        self.weapons = weapons  # mutable list — watch out!

    def clone(self):
        return copy.deepcopy(self)

    def __repr__(self):
        return f"Character({self.name}, hp={self.health}, weapons={self.weapons})"


# Create base template
base_warrior = Character("Warrior", health=100, armor=50, weapons=["sword", "shield"])

# Clone and customize — original untouched
warrior_2 = base_warrior.clone()
warrior_2.name = "Warrior-2"
warrior_2.weapons.append("axe")

print(base_warrior)  # Character(Warrior, hp=100, weapons=['sword', 'shield'])
print(warrior_2)     # Character(Warrior-2, hp=100, weapons=['sword', 'shield', 'axe'])
```

## 2. Shallow vs Deep Copy — The Critical Difference

```python
import copy

class Engine:
    def __init__(self, horsepower):
        self.horsepower = horsepower

class Car:
    def __init__(self, model, engine, features):
        self.model = model
        self.engine = engine      # nested object
        self.features = features  # mutable list

original = Car("Sedan", Engine(200), ["GPS", "Sunroof"])

# --- SHALLOW COPY ---
shallow = copy.copy(original)
shallow.model = "Coupe"
shallow.engine.horsepower = 999   # ⚠️ modifies ORIGINAL's engine too!
shallow.features.append("Turbo") # ⚠️ modifies ORIGINAL's list too!

print(original.engine.horsepower)  # 999  ← mutated!
print(original.features)           # ['GPS', 'Sunroof', 'Turbo'] ← mutated!

# --- DEEP COPY ---
original2 = Car("Sedan", Engine(200), ["GPS", "Sunroof"])
deep = copy.deepcopy(original2)
deep.engine.horsepower = 999
deep.features.append("Turbo")

print(original2.engine.horsepower)  # 200  ✅ safe
print(original2.features)           # ['GPS', 'Sunroof'] ✅ safe
```

```
Shallow Copy:
  original ──► engine (shared ref) ◄── clone
  original ──► features [ ] (shared) ◄── clone

Deep Copy:
  original ──► engine_A        clone ──► engine_B (independent)
  original ──► features_A [ ]  clone ──► features_B [ ] (independent)
```

## 3. Customizing Clone Behavior with `__copy__` and `__deepcopy__`

```python
import copy

class DatabaseConnection:
    def __init__(self, host, port, pool_size):
        self.host = host
        self.port = port
        self.pool_size = pool_size
        self._active_connections = []  # should NOT be cloned
        self._connect()

    def _connect(self):
        print(f"Connecting to {self.host}:{self.port}...")
        # Expensive operation!

    def __copy__(self):
        # Control exactly what gets shallow-copied
        cls = self.__class__
        new_obj = cls.__new__(cls)
        new_obj.__dict__.update(self.__dict__)
        new_obj._active_connections = []  # reset — don't share connections
        return new_obj

    def __deepcopy__(self, memo):
        # Control exactly what gets deep-copied
        cls = self.__class__
        new_obj = cls.__new__(cls)
        memo[id(self)] = new_obj
        for k, v in self.__dict__.items():
            if k == "_active_connections":
                setattr(new_obj, k, [])  # always reset
            else:
                setattr(new_obj, k, copy.deepcopy(v, memo))
        # Note: skips _connect() — no expensive reconnection!
        return new_obj


db = DatabaseConnection("localhost", 5432, pool_size=10)
db2 = copy.deepcopy(db)  # no reconnection cost, clean state
db2.port = 5433
```

## Real-World Example -> ML Model Config

```python
import copy
from dataclasses import dataclass, field

@dataclass
class ModelConfig:
    model_type: str = "transformer"
    layers: int = 12
    hidden_size: int = 768
    dropout: float = 0.1
    learning_rate: float = 3e-4
    optimizer: str = "adam"
    augmentations: list = field(default_factory=list)

    def clone(self, **overrides):
        new_config = copy.deepcopy(self)
        for k, v in overrides.items():
            setattr(new_config, k, v)
        return new_config


# One base config
base = ModelConfig(augmentations=["flip", "crop"])

# Many experiments — no repetition
small  = base.clone(layers=6,  hidden_size=256, learning_rate=1e-3)
medium = base.clone(layers=12, hidden_size=512)
large  = base.clone(layers=24, hidden_size=1024, dropout=0.05)
no_aug = base.clone(augmentations=[])  # independent list!

print(small)   # ModelConfig(layers=6,  hidden_size=256, ...)
print(large)   # ModelConfig(layers=24, hidden_size=1024, ...)
print(base.augmentations)  # ['flip', 'crop'] — untouched ✅
```
